st.set_page_config(
    page_title="English Quiz Generator",
    page_icon="📝",
    layout="wide",
    initial_sidebar_state="collapsed",
)

# ─── Session State Initialization ───
if "questions" not in st.session_state:
    st.session_state.questions = []
if "generated" not in st.session_state:
    st.session_state.generated = False

# ─── Header ───
st.markdown("""
<div class="main-header">
    <h1>📝 English Quiz Generator</h1>
    <p>教師用問題作成ツール — 画像から類似問題を生成してPowerPointへ出力</p>
</div>
""", unsafe_allow_html=True)

# ─── Sidebar: API Key ───
with st.sidebar:
    st.header("⚙️ 設定")
    api_key = st.text_input("Anthropic API Key", type="password",
                            help="Claude APIキーを入力してください。sk-ant-で始まるキーです。")
    st.markdown("---")
    st.markdown("**使い方**")
    st.markdown("""
    1. APIキーを入力
    2. 教師用資料の画像をアップロード
    3. 設定を調整して「生成」ボタン
    4. 問題を確認・編集
    5. PowerPointをダウンロード
    """)

# ─── Main Content ───
col_upload, col_settings = st.columns([3, 2])

with col_upload:
    st.subheader("📷 画像アップロード")
    uploaded_files = st.file_uploader(
        "教師用資料の問題と解答部分の画像をアップロードしてください",
        type=["png", "jpg", "jpeg", "gif", "webp"],
        accept_multiple_files=True,
        help="ドラッグ＆ドロップまたはクリックして選択"
    )

    if uploaded_files:
        cols = st.columns(min(len(uploaded_files), 4))
        for i, f in enumerate(uploaded_files):
            with cols[i % 4]:
                st.image(f, caption=f"画像 {i+1}", use_container_width=True)

with col_settings:
    st.subheader("⚙️ 生成設定")

    question_types = st.multiselect(
        "問題の種類",
        options=["穴埋め問題", "選択問題（4択）", "並べ替え問題"],
        default=["穴埋め問題", "選択問題（4択）", "並べ替え問題"],
    )

    num_questions = st.select_slider(
        "生成する問題数",
        options=[3, 4, 5],
        value=5,
    )

    difficulty = st.selectbox(
        "難易度",
        options=["やさしい（大学1年初級）", "ふつう（大学1年中級）", "むずかしい（大学1年上級）"],
        index=1,
    )

    include_answers = st.checkbox("解答スライドを含める", value=True)

st.markdown("---")

# ─── Generate Button ───
generate_col1, generate_col2 = st.columns([1, 4])
with generate_col1:
    generate_btn = st.button("🤖 類似問題を生成", type="primary", use_container_width=True,
                              disabled=not uploaded_files or not api_key)
with generate_col2:
    if not api_key:
        st.info("👈 サイドバーでAPIキーを入力してください")
    elif not uploaded_files:
        st.info("👆 画像をアップロードしてください")

# ─── Question Generation ───
if generate_btn and uploaded_files and api_key:
    with st.spinner("🔄 画像を分析中... Claude APIが問題を読み取り、類似問題を生成しています"):
        try:
            # Prepare images
            image_contents = []
            for f in uploaded_files:
                data = base64.b64encode(f.read()).decode("utf-8")
                f.seek(0)
                media_type = f.type or "image/png"
                image_contents.append({
                    "type": "image",
                    "source": {"type": "base64", "media_type": media_type, "data": data}
                })

            type_labels = "、".join(question_types)
            diff_map = {
                "やさしい（大学1年初級）": "初級",
                "ふつう（大学1年中級）": "中級",
                "むずかしい（大学1年上級）": "上級",
            }
            diff_label = diff_map.get(difficulty, "中級")

            prompt = f"""あなたは大学1年生向け英語必修科目の教師です。以下の画像は教師用資料に載っている問題と解答です。この画像を分析し、同じ形式・難易度で異なる内容の類似問題を{num_questions}問作成してください。

作成する問題の種類: {type_labels}
難易度: {diff_label}（大学1年生レベル）

重要な指示:
- 画像の問題と同じ形式（文法項目、語彙レベル、問題構造）を維持
- 問題の内容（使う単語・文章）は完全に新しいものにする
- 各問題には必ず正解を付ける
- 選択問題の場合は必ず4つの選択肢を含める
- 並べ替え問題の場合は並べ替える語句と正しい順序を示す

以下のJSON形式で出力。JSONのみ出力し他のテキストは含めない:
[{{"type":"FILL_BLANK"|"MULTIPLE_CHOICE"|"REORDER","question":"問題文","options":["A","B","C","D"],"answer":"正解","explanation":"解説（日本語）"}}]
optionsはMULTIPLE_CHOICEのみ。"""

            client = anthropic.Anthropic(api_key=api_key)
            message = client.messages.create(
                model="claude-sonnet-4-20250514",
                max_tokens=4000,
                messages=[{
                    "role": "user",
                    "content": image_contents + [{"type": "text", "text": prompt}]
                }]
            )

            text = "".join(block.text for block in message.content if hasattr(block, "text"))
            match = re.search(r"\[[\s\S]*\]", text)
            if not match:
                st.error("問題の生成に失敗しました。もう一度お試しください。")
            else:
                parsed = json.loads(match.group(0).replace("```json", "").replace("```", "").strip())
                if isinstance(parsed, list) and len(parsed) > 0:
                    st.session_state.questions = parsed
                    st.session_state.generated = True
                    st.success(f"✅ {len(parsed)}問の類似問題を生成しました！")
                else:
                    st.error("有効な問題が生成されませんでした。")

        except anthropic.AuthenticationError:
            st.error("❌ APIキーが無効です。正しいキーを入力してください。")
        except Exception as e:
            st.error(f"❌ エラーが発生しました: {str(e)}")

# ─── Display Questions ───
if st.session_state.questions:
    st.markdown("---")
    st.subheader(f"📋 生成された問題（{len(st.session_state.questions)}問）")

    TYPE_LABELS = {
        "FILL_BLANK": ("穴埋め", "fill"),
        "MULTIPLE_CHOICE": ("選択", "mc"),
        "REORDER": ("並べ替え", "reorder"),
    }

    for i, q in enumerate(st.session_state.questions):
        label, css_class = TYPE_LABELS.get(q.get("type", ""), ("問題", "fill"))

        with st.expander(f"Q{i+1} — {label}: {q['question'][:60]}...", expanded=True):
            st.markdown(f'<span class="q-badge {css_class}">{label}</span>', unsafe_allow_html=True)

            # Editable question
            new_q = st.text_area(f"問題文", value=q["question"], key=f"q_{i}", height=80)
            st.session_state.questions[i]["question"] = new_q

            # Options for MC
            if q.get("type") == "MULTIPLE_CHOICE" and q.get("options"):
                st.markdown("**選択肢:**")
                opt_cols = st.columns(2)
                for j, opt in enumerate(q["options"]):
                    with opt_cols[j % 2]:
                        is_answer = opt == q.get("answer", "")
                        prefix = "✅ " if is_answer else ""
                        st.markdown(f"{prefix}{chr(65+j)}. {opt}")

            # Editable answer
            new_a = st.text_input(f"正解", value=q.get("answer", ""), key=f"a_{i}")
            st.session_state.questions[i]["answer"] = new_a

            # Explanation
            if q.get("explanation"):
                st.caption(f"💡 {q['explanation']}")

    # ─── PowerPoint Generation ───
    st.markdown("---")
    st.subheader("📥 PowerPointをダウンロード")

    def generate_pptx(questions, show_answers):
        """Generate a PowerPoint file from questions."""
        prs = Presentation()
        prs.slide_width = Inches(10)
        prs.slide_height = Inches(5.625)

        def add_rect(slide, left, top, width, height, color):
            from pptx.util import Emu as E
            shape = slide.shapes.add_shape(
                1, left, top, width, height  # MSO_SHAPE.RECTANGLE = 1
            )
            shape.fill.solid()
            shape.fill.fore_color.rgb = color
            shape.line.fill.background()
            return shape

        def add_text_box(slide, left, top, width, height, text, font_size=14,
                         font_name="Calibri", color=RGBColor(0x2D, 0x37, 0x48),
                         bold=False, italic=False, alignment=PP_ALIGN.LEFT):
            txBox = slide.shapes.add_textbox(left, top, width, height)
            tf = txBox.text_frame
            tf.word_wrap = True
            p = tf.paragraphs[0]
            p.text = text
            p.font.size = Pt(font_size)
            p.font.name = font_name
            p.font.color.rgb = color
            p.font.bold = bold
            p.font.italic = italic
            p.alignment = alignment
            return txBox

        QUESTION_TYPE_LABELS = {
            "FILL_BLANK": "穴埋め問題",
            "MULTIPLE_CHOICE": "選択問題（4択）",
            "REORDER": "並べ替え問題",
        }

        # ── Title Slide ──
        slide = prs.slides.add_slide(prs.slide_layouts[6])  # Blank layout
        add_rect(slide, Inches(0), Inches(0), Inches(10), Inches(5.625), RGBColor(0x1A, 0x36, 0x5D))
        add_rect(slide, Inches(0), Inches(4.2), Inches(10), Inches(1.425), RGBColor(0x2D, 0x37, 0x48))
        add_text_box(slide, Inches(0.8), Inches(1.0), Inches(8.4), Inches(1.2),
                     "English Quiz", font_size=44, font_name="Georgia",
                     color=RGBColor(0xFF, 0xFF, 0xFF), bold=True)
        add_text_box(slide, Inches(0.8), Inches(2.5), Inches(8.4), Inches(0.6),
                     "類似問題 — 授業用", font_size=20,
                     color=RGBColor(0xCB, 0xD5, 0xE0))
        today = datetime.now()
        add_text_box(slide, Inches(0.8), Inches(3.3), Inches(8.4), Inches(0.4),
                     f"{today.year}年{today.month}月{today.day}日", font_size=14,
                     color=RGBColor(0xA0, 0xAE, 0xC0))

        # ── Question Slides (one per question) ──
        for i, q in enumerate(questions):
            slide = prs.slides.add_slide(prs.slide_layouts[6])

            # Header bar
            add_rect(slide, Inches(0), Inches(0), Inches(10), Inches(0.7), RGBColor(0x1A, 0x36, 0x5D))
            add_text_box(slide, Inches(0.5), Inches(0.05), Inches(9), Inches(0.6),
                         f"Question {i+1} / {len(questions)}", font_size=22,
                         font_name="Georgia", color=RGBColor(0xFF, 0xFF, 0xFF), bold=True)

            # Type badge
            type_label = QUESTION_TYPE_LABELS.get(q.get("type", ""), q.get("type", ""))
            add_text_box(slide, Inches(0.5), Inches(0.95), Inches(1.5), Inches(0.3),
                         type_label, font_size=9, color=RGBColor(0x71, 0x80, 0x96),
                         alignment=PP_ALIGN.CENTER)

            # Question text
            add_text_box(slide, Inches(0.5), Inches(1.4), Inches(9), Inches(1.2),
                         q["question"], font_size=16, color=RGBColor(0x2D, 0x37, 0x48))

            # Options (two-column layout)
            if q.get("type") == "MULTIPLE_CHOICE" and q.get("options"):
                opts = q["options"]
                half = (len(opts) + 1) // 2
                left_text = "\n".join(opts[:half])
                right_text = "\n".join(opts[half:])

                add_text_box(slide, Inches(0.8), Inches(2.8), Inches(4), Inches(half * 0.35),
                             left_text, font_size=14, color=RGBColor(0x4A, 0x55, 0x68))
                if right_text:
                    add_text_box(slide, Inches(5.2), Inches(2.8), Inches(4), Inches(len(opts[half:]) * 0.35),
                                 right_text, font_size=14, color=RGBColor(0x4A, 0x55, 0x68))

        # ── Answers Slide ──
        if show_answers:
            slide = prs.slides.add_slide(prs.slide_layouts[6])
            add_rect(slide, Inches(0), Inches(0), Inches(10), Inches(0.7), RGBColor(0x38, 0xA1, 0x69))
            add_text_box(slide, Inches(0.5), Inches(0.05), Inches(9), Inches(0.6),
                         "Answers", font_size=22, font_name="Georgia",
                         color=RGBColor(0xFF, 0xFF, 0xFF), bold=True)

            y = 1.0
            for i, q in enumerate(questions):
                # Answer line
                add_text_box(slide, Inches(0.5), Inches(y), Inches(1), Inches(0.35),
                             f"Q{i+1}.", font_size=14, bold=True,
                             color=RGBColor(0x1A, 0x36, 0x5D))
                add_text_box(slide, Inches(1.2), Inches(y), Inches(8.3), Inches(0.35),
                             q.get("answer", ""), font_size=12, bold=True,
                             color=RGBColor(0x38, 0xA1, 0x69))

                # Explanation
                if q.get("explanation"):
                    add_text_box(slide, Inches(1.2), Inches(y + 0.3), Inches(8.3), Inches(0.25),
                                 q["explanation"], font_size=10, italic=True,
                                 color=RGBColor(0x71, 0x80, 0x96))
                    y += 0.7
                else:
                    y += 0.45

        # Save to buffer
        buffer = io.BytesIO()
        prs.save(buffer)
        buffer.seek(0)
        return buffer

    dl_col1, dl_col2 = st.columns([1, 3])
    with dl_col1:
        pptx_buffer = generate_pptx(st.session_state.questions, include_answers)
        today = datetime.now()
        filename = f"English_Quiz_{today.strftime('%Y%m%d')}.pptx"

        st.download_button(
            label="📥 PowerPointをダウンロード",
            data=pptx_buffer,
            file_name=filename,
            mime="application/vnd.openxmlformats-officedocument.presentationml.presentation",
            type="primary",
            use_container_width=True,
        )

    with dl_col2:
        st.caption("問題スライド + 解答スライドが含まれたPowerPointファイルがダウンロードされます")

    # ─── JSON Export (backup) ───
    with st.expander("📄 問題データ（JSON）"):
        json_data = json.dumps({
            "questions": st.session_state.questions,
            "showAnswers": include_answers,
        }, ensure_ascii=False, indent=2)
        st.code(json_data, language="json")
        st.download_button(
            label="JSONとして保存",
            data=json_data,
            file_name=f"quiz_data_{today.strftime('%Y%m%d')}.json",
            mime="application/json",
        )
